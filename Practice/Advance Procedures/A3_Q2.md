```asm
include Irvine32.inc

.data
    arr dword 6, 1, 2, 4, 19, 0, 16, 3
    m1  byte "Write Integer to pass in param: ", 0

.code
; Proto declaration with stdcall
_FindMax proto stdcall,
    arr1: ptr dword,
    max: dword,
    idx: dword,
    user_val: dword

main proc stdcall
    mov edx, offset m1
    call WriteString
    call ReadInt

    invoke _FindMax, addr arr, 0, 0, eax

    call WriteInt
    call Crlf
    invoke ExitProcess, 0
main endp


_FindMax proc stdcall,
    arr1: ptr dword,
    max: dword,
    idx: dword,
    user_val: dword

    cmp idx,sizeof arr
    jge Return

    mov esi,arr1
    add esi,idx
    mov ebx,[esi]
    cmp max, ebx
    jge nextcall
    mov eax,ebx
    mov max, ebx
    nextcall:
    add idx,4
    invoke _FindMax,arr1,max,idx,user_val

    Return:
    ret
_FindMax endp

end main

```
