```asm
INCLUDE Irvine32.inc

.data
   s1 sbyte ?
   s2 sbyte ?
   u word ?
   m1 byte "Enter Signed 1 and 2 then unsigned number",0
.code
main PROC
    mov edx,offset m1
    call crlf
    call readint
    mov s1,al
    call readint
    mov s2,al
    call readint
    mov u,ax

    movsx eax,s1
    movsx ebx,s2
    movzx ecx,u

    xchg eax,ebx

    inc eax
    dec ebx

    mov edx,ebx
    add edx,eax

    sub eax,ebx
    mov edx,eax

    imul edx,ecx
    neg edx

    mov eax,edx
    call writeint

    

    exit
main ENDP
END main

```
