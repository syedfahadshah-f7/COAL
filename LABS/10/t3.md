### Task: 3
``` asm
include Irvine32.inc

.data
  m1 byte "Enter Integer for Cube ",0

.code

main proc 
   
    call LocalCube
    exit
main endp

LocalCube proc 
    enter 4,0
    mov edx,offset m1
    call writestring
    call readint
    mov dword ptr[ebp-4],eax

    mov ebx,eax
    mul ebx
    mul ebx
    call writeint
    leave
    ret              
LocalCube endp

end main

```
