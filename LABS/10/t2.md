```asm
include Irvine32.inc

.data
  arr1 dword 1,2,3,4,5,6,7,8,9,8,7,6,5,4,3,2

.code

main proc 
   
    push offset arr1
    call ArrayAvg
    exit
main endp

ArrayAvg proc 
    push ebp
    mov ebp, esp

    mov esi, [ebp+8]    ; arr1

    mov ecx,lengthof arr1
    mov ebx,ecx
    mov eax,0
    L1:
    add eax, dword ptr [esi]
    add esi,4
    loop L1

    xor edx,edx
    div ebx
    call WriteInt
    call Crlf           

    pop ebp
    ret 4             
ArrayAvg endp

end main
```
