### Task: 4
```asm
include Irvine32.inc

.data
  arr dword 5 DUP(?)
  m1 byte "Enter 5 integers: ",0
  m2 byte "Not all elements in array are even",0
  m3 byte "Smallest Even Number entered is : ",0

.code
main proc 
    mov esi, offset arr
    mov ecx, lengthof arr
    mov edx, offset m1
    call writestring

read_loop:
    call readint
    mov [esi], eax
    add esi, 4
    loop read_loop

    push offset arr
    call CheckEven

    exit
main endp

CheckEven proc
    push ebp
    mov ebp, esp
    mov esi, [ebp + 8]        
    mov ecx, lengthof arr
           
check_loop:
    mov eax, [esi]
    test eax, 1              
    jnz not_even
    add esi, 4
    loop check_loop

    ; all even
    push [ebp + 8]
    call SmallestEven
    jmp end_check

not_even:
    mov edx, offset m2
    call writestring
    call crlf

end_check:
    pop ebp
    ret 4
CheckEven endp

SmallestEven proc
    push ebp
    mov ebp, esp
    mov esi, [ebp + 8]
    mov ecx, lengthof arr
    mov ebx, 7FFFFFFFh        

find_loop:
    mov eax, [esi]
    cmp eax, ebx
    jge skip
    mov ebx, eax

skip:
    add esi, 4
    loop find_loop

    mov edx, offset m3
    call writestring
    mov eax, ebx
    call writeint
    call crlf

    pop ebp
    ret 4
SmallestEven endp

end main

```
