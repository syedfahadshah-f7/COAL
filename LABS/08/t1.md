### Task: 1
```asm
include irvine32.inc
.data
	m1 byte "Intgers are equal",0
	m2 byte "Intgers are not equal",0
	m3 byte "Enter 4 integers",0
	vals dword 4 DUP(?)
.code
main proc
	mov edx,offset m3
	call writestring
	call crlf
	mov esi,0
	mov ecx,lengthof vals
	L1:
	call readint
	mov vals[esi* type vals],eax
	inc esi;
	loop L1

	mov esi,1
	mov ecx,lengthof vals -1
	mov eax,vals[0]
	L2:
	cmp eax,vals[esi* type vals]
	je cont
	mov edx, offset m2
	call writestring
	cll crlf
	exit
	cont:
	inc esi;
	loop L2

   exit
main endp
end main

```
